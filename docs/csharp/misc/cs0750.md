## <a name="example"></a><span data-ttu-id="b4368-108">예제</span><span class="sxs-lookup"><span data-stu-id="b4368-108">Example</span></span>  

 <span data-ttu-id="b4368-109">다음 예제에서는 CS0750을 생성합니다.</span><span class="sxs-lookup"><span data-stu-id="b4368-109">The following example generates CS0750:</span></span>  
  
```csharp  
// cs0750.cs  
using System;  
  
public class Base  
{  
    protected virtual void PartG()  
    {  
    }  
  
    protected void PartH()  
    {  
    }  
    protected virtual void PartI()  
    {  
    }  
}  
  
public partial class C:Base  
{  
    // All these partial method declarations  
    // will generate CS0750.  
    public partial void PartA();  
    private partial void PartB();  
    protected partial void PartC();  
    internal partial void PartD();  
    virtual partial void PartE();  
    abstract partial void PartF();  
    override partial void PartG();  
    new partial void PartH();  
    sealed override partial void PartI();  
    extern partial void PartJ();  
  
    public static int Main()  
    {  
        return 1;  
    }  
}  
```
